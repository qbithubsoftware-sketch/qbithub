    77→/// Master record for a downloadable file.  Only metadata + a secure storage
    78→/// path are stored here — the binary itself lives in cloud storage.
    79→model Download {
    80→  id                  String           @id @default(cuid())
    81→  productId           String? // reference to the product this download belongs to
    82→  name                String // "Universal Thermal Printer Driver"
    83→  version             String // "v2.4.1"
    84→  categoryId          String
    85→  category            DownloadCategory @relation(fields: [categoryId], references: [id])
    86→  releaseDate         DateTime
    87→  fileSize            Int // bytes
    88→  storagePath         String // secure cloud-storage key (never exposed to client)
    89→  checksum            String? // SHA-256 hash for integrity verification
    90→  visibility          String           @default("public") // "public" | "internal" | "restricted"
    91→  featured            Boolean          @default(false)
    92→  latest              Boolean          @default(false)
    93→  downloadCount       Int              @default(0)
    94→  description         String?
    95→  installInstructions String?
    96→  knownIssues         String?
    97→  supportedProducts   String? // JSON string array of product names/models
    98→  createdAt           DateTime         @default(now())
    99→  updatedAt           DateTime         @updatedAt
   100→
   101→  operatingSystems DownloadOS[]
   102→  releaseNotes     ReleaseNote[]

   107→  @@index([categoryId])
