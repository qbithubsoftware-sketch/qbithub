    69→/// Master record for a downloadable file.  Only metadata + a secure storage
    70→/// path are stored here — the binary itself lives in cloud storage.
    71→model Download {
    72→  id            String   @id @default(cuid())
    73→  productId     String?  // reference to the product this download belongs to
    74→  name          String   // "Universal Thermal Printer Driver"
    75→  version       String   // "v2.4.1"
    76→  categoryId    String
    77→  category      DownloadCategory @relation(fields: [categoryId], references: [id])
    78→  releaseDate   DateTime
    79→  fileSize      Int      // bytes
    80→  storagePath   String   // secure cloud-storage key (never exposed to client)
    81→  checksum      String?  // SHA-256 hash for integrity verification
    82→  visibility    String   @default("public") // "public" | "internal" | "restricted"
    83→  featured      Boolean  @default(false)
    84→  latest        Boolean  @default(false)
    85→  downloadCount Int      @default(0)
    86→  description   String?
    87→  installInstructions String?
    88→  knownIssues   String?
    89→  supportedProducts String? // JSON string array of product names/models
    90→  createdAt     DateTime @default(now())
    91→  updatedAt     DateTime @updatedAt
    92→
    93→  operatingSystems DownloadOS[]
    94→  releaseNotes    ReleaseNote[]
    95→  downloadHistory DownloadHistory[]
    96→  favorites       FavoriteDownload[]
    97→
    98→  @@index([categoryId])
    99→  @@index([visibility])
   100→  @@index([featured])
   101→  @@index([latest])
