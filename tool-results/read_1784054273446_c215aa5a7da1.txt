   115→function getYouTubeId(url: string): string | null {
   116→  const patterns = [
   117→    /(?:youtube\.com\/watch\?v=|youtu\.be\/|youtube\.com\/embed\/|youtube\.com\/v\/)([A-Za-z0-9_-]{11})/,
   118→  ];
   119→  for (const p of patterns) {
   120→    const m = url.match(p);
   121→    if (m) return m[1];
   122→  }
   123→  return null;
   124→}

   126→export function ProductDetailClient({ product }: { product: ProductDetail }) {
   127→  const [activeImage, setActiveImage] = useState(0);
   128→  const [copied, setCopied] = useState(false);
   129→  const [activeTab, setActiveTab] = useState<"specs" | "features" | "downloads" | "videos">("specs");
   130→
   131→  // Build gallery: prefer product.galleryImages, fall back to imageUrl, fall back to icon placeholder.
   132→  const gallery = useMemo(() => {
   133→    const imgs: { url: string; alt: string }[] = [];

   135→      imgs.push(...product.galleryImages);
   136→    } else if (product.imageUrl) {
   137→      imgs.push({ url: product.imageUrl, alt: product.name });
   138→    }
   139→    // Also include any media files of type "image"
   140→    for (const m of product.mediaFiles) {
   141→      if (m.type === "image" && !imgs.some((i) => i.url === m.url)) {
   142→        imgs.push({ url: m.url, alt: m.altText ?? product.name });
   143→      }
   144→    }
   145→    return imgs;
   146→  }, [product]);
   147→
   148→  // Group specifications by `group` field (if present).
   149→  const specGroups = useMemo(() => {
   150→    const groups: Record<string, { property: string; value: string }[]> = {};
   151→    for (const s of product.specifications) {
   152→      const g = s.group ?? "General";
   153→      if (!groups[g]) groups[g] = [];
   154→      groups[g].push({ property: s.property, value: s.value });
   155→    }
   156→    return Object.entries(groups);
   157→  }, [product.specifications]);
   158→

   160→  const availableDownloads = DOWNLOAD_TYPES.filter((d) => {

   167→

   169→    if (typeof window !== "undefined") {
   170→      window.open(url, "_blank", "noopener,noreferrer");

   172→  };

   175→    const shareUrl = typeof window !== "undefined" ? window.location.href : `https://hub.qbit.com/products/${product.slug}`;

   177→      try {
   178→        await navigator.share({ title: product.name, text: product.description ?? "", url: shareUrl });
   179→        return;
   180→      } catch { /* fall through to clipboard */ }
   181→    }
   182→    if (typeof navigator !== "undefined" && navigator.clipboard) {
   183→      await navigator.clipboard.writeText(shareUrl);
   184→      setCopied(true);
   185→      setTimeout(() => setCopied(false), 2000);
   186→    }
   187→  };
   188→
   189→  const currentImage = gallery[activeImage];
   190→  const iconName = DEVICE_TYPE_ICON[product.deviceType] ?? "inventory_2";
   191→  const publicUrl = `https://hub.qbit.com/products/${product.slug}`;
   192→  const qrApiUrl = `https://api.qrserver.com/v1/create-qr-code/?size=240x240&data=${encodeURIComponent(publicUrl)}`;
   193→
   194→  return (
   195→    <div className="min-h-screen bg-qbit-surface">
   196→      {/* Top bar */}

   198→        <div className="mx-auto flex max-w-7xl items-center gap-3 px-4 py-3">
   199→          <Link
   200→            href="/products"
   201→            className="flex items-center gap-2 text-sm font-medium text-qbit-on-surface-variant hover:text-qbit-primary"
   202→          >
   203→            <Icon name="arrow_back" className="text-[20px]" />
   204→            All Products
   205→          </Link>
   206→          <span className="text-qbit-outline-variant">/</span>
   207→          {product.category && (
   208→            <>
   209→              <Link
   210→                href={`/products?category=${product.category}`}
   211→                className="text-sm font-medium text-qbit-on-surface-variant hover:text-qbit-primary capitalize"
   212→              >
   213→                {product.category.replace(/-/g, " ")}
   214→              </Link>
   215→              <span className="text-qbit-outline-variant">/</span>
   216→            </>
   217→          )}
   218→          <span className="truncate text-sm font-semibold text-qbit-on-surface">{product.name}</span>
   219→        </div>
   220→      </header>
   221→
   222→      <main className="mx-auto max-w-7xl px-4 py-8">
   223→        {/* ------------------- HERO ------------------- */}
   224→        <section className="grid grid-cols-1 gap-8 lg:grid-cols-12">
   225→          {/* Gallery */}

   227→            <div className="group relative aspect-[4/3] overflow-hidden rounded-2xl border border-qbit-outline-variant bg-white shadow-sm">
   228→              {currentImage ? (
   229→                // eslint-disable-next-line @next/next/no-img-element
   230→                <img
   231→                  src={currentImage.url}
   232→                  alt={currentImage.alt}
   233→                  className="h-full w-full object-cover transition-transform duration-500 group-hover:scale-105"
   234→                />
   235→              ) : (
   236→                <div className="flex h-full w-full items-center justify-center bg-gradient-to-br from-qbit-surface-container-low via-white to-qbit-surface-container-high">
   237→                  <Icon name={iconName} className="text-[120px] text-qbit-primary/70" filled />

   280→                <StatusBadge
   281→                  variant={product.status === "active" ? "success" : "neutral"}
   282→                  dot
   283→                >
   284→                  {product.status === "active" ? "Active" : product.status}

   286→              </div>
   287→              <h1 className="text-3xl font-bold tracking-tight text-qbit-on-surface md:text-4xl">

   291→                <span><strong className="text-qbit-on-surface">Model:</strong> {product.model}</span>

   333→                  AI Diagnostics Supported
   334→                </div>
   335→              )}

   338→            {/* OS support */}

   486→                      </div>
   487→                    </SurfaceCard>
   488→                  );
   489→                })}
   490→
   491→                {/* Firmware card (always shown if latestFirmwareVersion is set) */}
   492→                {hasFirmwareCard && (
   493→                  <SurfaceCard className="flex items-start gap-4 p-5">

   508→                      </Link>
   509→                    </div>
   510→                  </SurfaceCard>

   513→                {/* Media-file-backed downloads (driver/firmware/sdk/utility/brochure/datasheet/warranty/manual/video) */}
   514→                {product.mediaFiles
   515→                  .filter((m) => ["driver", "firmware", "sdk", "utility", "brochure", "datasheet", "warranty", "manual"].includes(m.type))
   516→                  .filter((m) => !availableDownloads.some((d) => product[d.urlField] === m.url))
   517→                  .map((m) => (

   526→                          onClick={() => handleDownload(m.url, m.title)}
   527→                          className="mt-3 inline-flex items-center gap-1 text-xs font-semibold text-qbit-primary hover:underline"
   528→                        >
   529→                          Download / Open
   530→                          <Icon name="arrow_forward" className="text-[14px]" />
   531→                        </button>
   532→                      </div>
   533→                    </SurfaceCard>
   534→                  ))}
   535→              </div>
   536→
   537→              {availableDownloads.length === 0 && !hasFirmwareCard && product.mediaFiles.length === 0 && (
   538→                <EmptyState icon="download" message="No downloads available for this product yet." />
   539→              )}
   540→            </div>
   541→          )}
   542→
   543→          {/* Videos */}
   544→          {activeTab === "videos" && (

   569→                                target="_blank"
   570→                                rel="noopener noreferrer"
   571→                                className="flex flex-col items-center gap-2 text-qbit-primary"

   573→                                <Icon name="play_circle" className="text-[64px]" filled />

   575→                              </a>
   576→                            </div>
   577→                          </div>

   579→                        <div className="p-4">
   580→                          <h4 className="text-sm font-bold text-qbit-on-surface">{v.title}</h4>
   581→                        </div>
   582→                      </SurfaceCard>
   583→                    );
   584→                  })}
   585→                </div>
   586→              ) : (
   587→                <EmptyState icon="videocam" message="No video available." />
   588→              )}
   589→            </div>
   590→          )}
   591→        </section>
   592→
   593→        {/* ------------------- LONG DESCRIPTION ------------------- */}
   594→        {product.longDescription && (

   596→            <SurfaceCard className="p-6">
   597→              <h2 className="mb-3 text-xl font-bold text-qbit-on-surface">About this product</h2>
   598→              <p className="whitespace-pre-line text-sm leading-relaxed text-qbit-on-surface-variant">
   599→                {product.longDescription}
   600→              </p>
   601→            </SurfaceCard>
   602→          </section>
   603→        )}
   604→
   605→        {/* ------------------- TAGS + COMPATIBLE DEVICES ------------------- */}
   606→        {(product.tags.length > 0 || product.compatibleDevices.length > 0) && (
   607→          <section className="mt-8 grid grid-cols-1 gap-6 md:grid-cols-2">
   608→            {product.tags.length > 0 && (
   609→              <SurfaceCard className="p-6">
   610→                <h3 className="mb-3 text-sm font-bold uppercase tracking-wider text-qbit-on-surface-variant">Tags</h3>

   615→                </div>
   616→              </SurfaceCard>
   617→            )}

   619→              <SurfaceCard className="p-6">
   620→                <h3 className="mb-3 text-sm font-bold uppercase tracking-wider text-qbit-on-surface-variant">Compatible Devices</h3>

   623→                    <span
   624→                      key={d}

   626→                    >
   627→                      <Icon name="check_circle" className="text-[14px] text-qbit-success" />
   628→                      {d}
   629→                    </span>
   630→                  ))}
   631→                </div>
   632→              </SurfaceCard>
   633→            )}
   634→          </section>
   635→        )}
   636→
   637→        {/* ------------------- QR CODE + SHARE ------------------- */}
   638→        <section className="mt-8">
   639→          <SurfaceCard className="flex flex-col items-center gap-6 p-6 md:flex-row md:items-start">
   640→            <div className="flex flex-col items-center gap-2">
   641→              <div className="rounded-xl border border-qbit-outline-variant bg-white p-3">
   642→                {/* eslint-disable-next-line @next/next/no-img-element */}
   643→                <img src={qrApiUrl} alt={`QR code for ${product.name}`} width={180} height={180} />
   644→              </div>
   645→              <p className="text-xs font-medium text-qbit-on-surface-variant">Scan to view on mobile</p>
   646→            </div>
   647→            <div className="flex-1">
   648→              <h3 className="text-lg font-bold text-qbit-on-surface">Share this product</h3>
   649→              <p className="mt-1 text-sm text-qbit-on-surface-variant">
   650→                Send the direct link to colleagues or customers — they&apos;ll land right here.
   651→              </p>
   652→              <div className="mt-4 flex flex-wrap gap-3">
   653→                <input
   654→                  type="text"
   655→                  readOnly
   656→                  value={publicUrl}
   657→                  className="flex-1 rounded-lg border border-qbit-outline-variant bg-qbit-surface-container-low px-3 py-2 text-sm text-qbit-on-surface-variant"
   658→                />

   660→                  {copied ? "Copied" : "Copy Link"}
